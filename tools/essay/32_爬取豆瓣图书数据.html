  6. 爬取数据并解析

    package com.it.testjdbc.service;


import com.alibaba.fastjson.JSON;
import com.it.testjdbc.domain.Book;
import com.it.testjdbc.mapper.BookMapper;
import okhttp3.OkHttpClient;
import okhttp3.Request;
import okhttp3.Response;
import okhttp3.ResponseBody;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;
import java.util.UUID;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

@Service
public class BookService {


    @Autowired
    private BookMapper bookMapper;

    public void pullData() {
        OkHttpClient okHttpClient = new OkHttpClient();
        Request request = new Request.Builder().url("https://book.douban.com/tag/%E5%90%8D%E8%91%97").build();
        try (Response response = okHttpClient.newCall(request).execute()) {
            ResponseBody responseBody = response.body();
            String content = responseBody.string();
            content = getBody(content);
            getUl(content);

        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    /**
     * 获取图书列表信息
     *
     * @param content 网页html
     */
    void getUl(String content) {

        //String regex = "<ul class=\"subject-list\">([\\s\\S]*?)</ul>";
        String regex = "<li class=\"subject-item\">([\\s\\S]*?)</li>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        List<String> liInfo = new ArrayList<>();
        while (matcher.find()) {
            // 将图书列表信息暂存在liInfo中
            liInfo.add(replaceLine(matcher.group(1)));
        }

        List<Book> bookList = new ArrayList<>();
        // 循环处理图书列表信息
        for (String liContent : liInfo) {
            Book book = new Book();
            book.setId(UUID.randomUUID().toString());
            getInfo(liContent, book);
            getPictureUrl(liContent, book);
            bookList.add(book);
            //bookMapper.insert(book);
        }
        System.out.println("bookList.size() = " + bookList.size());
        System.out.println(JSON.toJSONString(bookList));
        // 将图书信息存储到数据库中
        bookMapper.addBookByBatch(bookList);
    }

    /**
     * 获取图书具体信息
     *
     * @param content
     * @param book
     */
    void getInfo(String content, Book book) {
        String regex = "<div class=\"info\">(.*)</div>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return;
        }
        content = matcher.group(1);
        // 获取图书url链接
        getBookUrl(content, book);
        // 获取图书发布机构信息
        getPublishInfo(content, book);
        // 获取图书评论信息
        getCommentInfo(content, book);
        // 获取图书简介
        getDetail(content, book);
        // 获取图书购买信息
        getBuyInfo(content, book);
    }

    /**
     * 获取图书购买信息
     *
     * @param content
     * @param book
     */
    void getBuyInfo(String content, Book book) {
        String regex = " <div class=\"ft\">(.*)</div>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (matcher.find()) {
            content = matcher.group(1);
        } else {
            return;
        }

        regex = "<span class=\"buy-info\">(.*)</span>";
        pattern = Pattern.compile(regex);
        matcher = pattern.matcher(content);
        if (matcher.find()) {
            content = matcher.group(1);
        } else {
            return;
        }

        // 获取图书够慢链接
        regex = "<a.*?href=[\"']?((https?://)?/?[^\"']+)[\"']?.*?>(.+)</a>";
        pattern = Pattern.compile(regex);
        matcher = pattern.matcher(content);
        if (matcher.find()) {
            book.setBuyUrl(matcher.group(1));
        }

        // 获取图书购买价格
        String priceString = matcher.group(3);
        regex = "([\\u4e00-\\u9fa5]*[\\s])([\\d]*.[\\d]*)([\\u4e00-\\u9fa5]*)";
        pattern = Pattern.compile(regex);
        matcher = pattern.matcher(priceString.trim());
        if (matcher.find()) {
            book.setBuyPrice(Double.parseDouble(matcher.group(2)));
        }
    }

    /**
     * 获取图书介绍信息
     *
     * @param content
     * @param book
     */
    void getDetail(String content, Book book) {
        String regex = "<p>([\\s\\S]*?)</p>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (matcher.find()) {
            book.setDetail(matcher.group(1).trim());
        }
    }

    /**
     * 获取图书评论信息
     *
     * @param content
     * @param book
     */
    void getCommentInfo(String content, Book book) {
        String regex = "<div class=\"star clearfix\">([\\s\\S]*?)</div>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return;
        }
        content = matcher.group(1);
        regex = "<span class=\"[\\w]*\">([\\s\\S]*?)</span>";
        pattern = Pattern.compile(regex);
        matcher = pattern.matcher(content);
        List<String> matchResult = new ArrayList<>();
        while (matcher.find()) {
            String temp = matcher.group(1).trim();
            if (temp.trim().length() != 0) {
                matchResult.add(temp.trim());
            }
        }
        book.setRatingNums(Double.parseDouble(matchResult.get(0)));
        String commentNumber = matchResult.get(1);
        book.setCommentNumber(Integer.parseInt(commentNumber.substring(commentNumber.indexOf("(") + 1, commentNumber.lastIndexOf("人"))));
    }

    /**
     * 获取图书发布信息
     *
     * @param content
     * @param book
     */
    void getPublishInfo(String content, Book book) {
        String regex = "<div class=\"pub\">([\\s\\S]*?)</div>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return;
        }
        String result = matcher.group(1);
        String[] split = result.split("/");
        if (split.length == 4) {
            //[清] 曹雪芹 著、高鹗 续 / 人民文学出版社 / 1996-12 / 59.70元
            book.setAuthor(split[0].trim());
            book.setPublish(split[1].trim());
            book.setPublishTime(split[2].trim());
            String priceString = split[3].trim();
            boolean b = priceString.contains("元");
            if (b) {
                book.setPrice(Double.parseDouble(priceString.substring(0, priceString.lastIndexOf("元"))));
            } else {
                book.setPrice(Double.parseDouble(priceString.trim()));
            }
        } else if (split.length == 5) {
            //[哥伦比亚] 加西亚·马尔克斯 / 范晔 / 南海出版公司 / 2011-6 / 39.50元
            book.setAuthor(split[0].trim());
            book.setTranslator(split[1].trim());
            book.setPublish(split[2].trim());
            book.setPublishTime(split[3].trim());
            String priceString = split[4].trim();
            boolean b = priceString.contains("元");
            if (b) {
                book.setPrice(Double.parseDouble(priceString.substring(0, priceString.lastIndexOf("元"))));
            } else {
                book.setPrice(Double.parseDouble(priceString));
            }
        }
    }

    /**
     * 获取图书链接信息
     *
     * @param content
     * @param book
     */
    void getBookUrl(String content, Book book) {
        String regex = "<h2 class=\"\">([\\s\\S]*?)</h2>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return;
        }
        content = matcher.group(1);

        regex = "<a.*?href=[\"']?((https?://)?/?[^\"']+)[\"']?.*?>(.+)</a>";
        pattern = Pattern.compile(regex);
        matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return;
        }
        book.setUrl(matcher.group(1));
        book.setName(matcher.group(3).trim());
    }

    /**
     * 替换文本中的换行符
     *
     * @param content
     * @return
     */
    String replaceLine(String content) {
        Pattern pattern = Pattern.compile("\n");
        Matcher matcher = pattern.matcher(content);
        if (matcher.find()) {
            content = matcher.replaceAll("");
        }
        return content;
    }

    /**
     * 获取图书的图片链接
     *
     * @param content
     * @param book
     */
    void getPictureUrl(String content, Book book) {
        String regex = "<div class=\"pic\">([\\s\\S]*?)</div>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return;
        }
        content = matcher.group(1);

        //获取图片链接信息
        regex = "<img class=\"\" src=\"([^\"]*?)\" .*>";
        pattern = Pattern.compile(regex);
        matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return;
        }
        book.setPicture(matcher.group(1));
    }

    /**
     * 从网页中提取body数据
     *
     * @param content
     * @return
     */
    String getBody(String content) {
        if (content == null || content.trim().length() == 0) {
            return content;
        }
        String regex = "<body>([\\s\\S]*?)</body>";
        Pattern pattern = Pattern.compile(regex);
        Matcher matcher = pattern.matcher(content);
        if (!matcher.find()) {
            return null;
        }
        content = matcher.group(1);
        return content;
    }
}
