# 从MongoDB中下载附件

public void getFile(@NotBlank(message = "id is null") @RequestParam("id") String id,
						HttpServletResponse response) throws Exception {
		ServletOutputStream outputStream = response.getOutputStream();
		String extension = FilenameUtils.getExtension(id);
		if ("gltf".equalsIgnoreCase(extension)) {
			response.addHeader("Accept-Ranges", "bytes");
			response.addHeader("Content-Type", "application/octet-stream");
			response.setHeader("Content-Disposition", "attachment; filename=" + id);
		} else {
			if (Tools.isNotNull(extension) && "svg".equalsIgnoreCase(extension)) {
				response.setContentType("image/svg+xml");
			} else {
				response.setContentType("image/jpeg");
			}
		}
		mongoDBService.getFile(id, outputStream);
		outputStream.flush();
	}
