# 批量更新某一文件夹下的文件名

import java.io.File;

public class RenameFile {
	public static void main(String[] args) {

		String basePath = "F:\\test";
		File fileDirectory = new File(basePath);
		if (fileDirectory.isDirectory()) {
			File[] listFiles = fileDirectory.listFiles();
			if (listFiles != null && listFiles.length != 0) {
				for (int i = 0; i < listFiles.length; i++) {
					File oldFile = listFiles[i];
					String oldFileName = oldFile.getName();
					int index = oldFileName.indexOf("_");
					String substring = oldFileName.substring(index + 1);
					String newFileName = basePath + "\\" + substring;
					boolean b = oldFile.renameTo(new File(newFileName));
					if (b) {
						System.out.println("更名成功");
					} else {
						System.out.println("更名失败");
					}
				}
			}
		}
	}
}
