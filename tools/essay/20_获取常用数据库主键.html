# 获取常用数据库主键

# 一、MYSQL

  1. sql

    select column_name
from information_schema.KEY_COLUMN_USAGE
where TABLE_SCHEMA = 'database_name'
  and TABLE_NAME = 'table_name'
  and CONSTRAINT_NAME = 'PRIMARY';

  2. 程序

    public String getPrimaryKry(String tableName, String initDatabase, Connection connection) throws Exception {
    // 获取主键
    String sql = "SELECT column_name FROM INFORMATION_SCHEMA.`KEY_COLUMN_USAGE` WHERE  table_name='%s' AND table_schema='%s' AND constraint_name='PRIMARY'";
    sql = String.format(PRIMARY_KRY_QUERY, tableName, initDatabase);
    try (ResultSet resultSet = connection.prepareStatement(sql).executeQuery();) {
        String primaryKey = null;
        if (resultSet.next()) {
            primaryKey = resultSet.getString("column_name");
        }
        return primaryKey;
    }
}

# 二、ORACLE

  1. sql

    select COLUMN_NAME
from ALL_CONSTRAINTS con,
     ALL_CONS_COLUMNS col
where con.CONSTRAINT_NAME = col.CONSTRAINT_NAME
  and con.CONSTRAINT_TYPE = 'P'
  and con.TABLE_NAME = 'table_name';

  2. 程序

    public String getPrimaryKry(String tableName, String initDatabase, Connection connection) throws Exception {
	String sql = "select COLUMN_NAME from ALL_CONSTRAINTS con, ALL_CONS_COLUMNS col where con.CONSTRAINT_NAME = col.CONSTRAINT_NAME and con.TABLE_NAME ='%s' and con.CONSTRAINT_TYPE = 'P'";
    sql = String.format(PRIMARY_KRY_QUERY, tableName);
	try (ResultSet resultSet = connection.prepareStatement(sql).executeQuery()) {
		String primaryKey = null;
		if (resultSet.next()) {
			primaryKey = resultSet.getString("column_name");
			System.out.println("primaryKey = " + primaryKey);
		}
		return primaryKey;
	}
}

# 三、SQLServer

  1. sql

    select column_name
from information_schema.key_column_usage
where constraint_catalog = 'database_name'
	and table_name = 'table_name';

  2. 程序

    public String getPrimaryKry(String tableName, String initDatabase, Connection connection) throws Exception {
    String sql = "select COLUMN_NAME from INFORMATION_SCHEMA.KEY_COLUMN_USAGE where CONSTRAINT_CATALOG = '%s' and TABLE_NAME = '%s'";
    sql = String.format(PRIMARY_KRY_QUERY, initDatabase, tableName);
    try (ResultSet resultSet = connection.prepareStatement(sql).executeQuery();) {
        String primaryKey = null;
        if (resultSet.next()) {
            primaryKey = resultSet.getString("column_name");
        }

        return primaryKey;
    }
}

# 四、PostgreSQL

  1. sql

    select pg_attribute.attname as colname
from pg_constraint
    inner join pg_class on pg_constraint.conrelid = pg_class.oid
    inner join pg_attribute on pg_attribute.attrelid = pg_class.oid
        and pg_attribute.attnum = pg_constraint.conkey[1]
    inner join pg_type on pg_type.oid = pg_attribute.atttypid
where pg_class.relname = 'table_name'
	and pg_constraint.contype = 'p';

  2. 程序

    public String getPrimaryKry(String tableName, String initDatabase, Connection connection) throws Exception {
    // 获取主键
    String sql = "select pg_attribute.attname as colname from " +
            "pg_constraint inner join pg_class on pg_constraint.conrelid = pg_class.oid inner " +
            "join pg_attribute on pg_attribute.attrelid = pg_class.oid " +
            " and pg_attribute.attnum = pg_constraint.conkey[1] inner join pg_type on pg_type.oid" +
            " = pg_attribute.atttypid " +
            " where pg_class.relname = '%s' and pg_constraint.contype = 'p'";
    sql = String.format(PRIMARY_KRY_QUERY, tableName);
    try (ResultSet resultSet = connection.prepareStatement(sql).executeQuery();) {
        String primaryKey = null;
        if (resultSet.next()) {
            primaryKey = resultSet.getString("colname");
        }
        return primaryKey;
    }
}
