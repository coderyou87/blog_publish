# 在同一台Linux中安装多个mysql

# 一、安装第一台MySQL

  1. 上传mysql安装包到/opt/software目录下

  2. 创建mysql用户组与用户

    groupadd mysql
useradd -g mysql mysql

  3. 解压mysql

    tar -zxvf /opt/software/mysqlXXX -C /opt/module/

  4. 初始化mysql

    /opt/module/mysqlXXX/bin/mysqld --initialize-insecure --user=mysql --basedir=/opt/module/mysqlXXX --datadir=/opt/module/mysqlXXX/data

  5. 安装mysql加密连接

    /opt/module/mysqlXXX/bin/mysql_ssl_rsa_setup --datadir=/opt/module/mysqlXXX/data

  6. 创建配置文件

    mkdir /opt/module/mysqlXXX/tmp
mkdir /opt/module/mysqlXXX/conf
vim /opt/module/mysqlXXX/conf/my.cnf

    编辑/opt/module/mysqlXXX/conf/my.cnf

    [client]
port = 3306
default-character-set=utf8
socket=/opt/module/mysqlXXX/tmp/mysql.sock
[mysqld]
# 一般配置选项
basedir = /usr/local/mysql
datadir = /usr/local/mysql/data
port = 3306
character-set-server=utf8
default_storage_engine=InnoDB
sql_mode=STRICT_TRANS_TABLES,NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTION
socket=/opt/module/mysqlXXX/tmp/mysql.sock
pid-file=/opt/module/mysqlXXX/tmp/mysqld.pid

  7. 启动mysql服务

    /opt/module/mysqlXXX/bin/mysqld --defaults-file=/opt/module/mysqlXXX/conf/my.cnf --user=root

  8. 启动mysql客户端

    /opt/module/mysqlXXX/bin/mysql --defaults-file=/opt/module/mysqlXXX/conf/my.cnf -u root -p

    /opt/module/mysqlXXX/bin/mysql -S /opt/module/mysqlXXX/tmp/mysql.sock -u root -p

  9. 修改初始化密码

    set password=password("New Password")

  10. 授权远程访问

    grant all privileges on *.* to 'root'@'%' identified by 'Your Password' with grant option;

    刷新权限

    flush privileges;

  11. 开放端口

    firewall-cmd --zone=public --add-port=XXXX/tcp --permanent
firewall-cmd --reload

  12. 编写服务启动脚本

    vim /usr/local/sbin/start_mysql_1

    #!/bin/bash
/opt/module/mysql/bin/mysqld --defaults-file=/opt/module/mysql/conf/my.cnf --user=root &

    chmod +x /usr/local/sbin/start_mysql_1

  13. 停止mysql

    pkill mysql
