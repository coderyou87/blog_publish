# SortedTest

package com.it.redisdemo;

import org.junit.After;
import org.junit.Before;
import org.junit.FixMethodOrder;
import org.junit.Test;
import org.junit.jupiter.api.DisplayName;
import org.junit.runners.MethodSorters;
import redis.clients.jedis.Jedis;
import redis.clients.jedis.JedisPool;
import redis.clients.jedis.JedisPoolConfig;
import redis.clients.jedis.Tuple;

import java.util.Arrays;
import java.util.HashMap;
import java.util.Map;
import java.util.Set;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertTrue;

/**
 * @program redisdemo
 * @description: SortedSetTest
 * @author: 游元博
 * @create: 2020/04/28 09:39
 */
@DisplayName("sorted_set test")
@FixMethodOrder(MethodSorters.NAME_ASCENDING)
public class SortedSetTest {

    private JedisPool jedisPool;

    private Jedis jedis;

    @Before
    @DisplayName("init")
    public void init() {
        JedisPoolConfig jedisPoolConfig = new JedisPoolConfig();
        jedisPoolConfig.setMaxTotal(10);
        jedisPoolConfig.setMaxIdle(10);
        jedisPoolConfig.setMinIdle(10);
        jedisPoolConfig.setMaxWaitMillis(-1);
        jedisPool = new JedisPool(jedisPoolConfig, "49.234.79.63", 7101, 0, null, 0);
        jedis = jedisPool.getResource();
    }

    @After
    @DisplayName("close")
    public void close() {
        if (jedis != null) {
            jedis.close();
        }
    }

    @Test
    @DisplayName("zadd")
    public void test_01_zadd() {
        Map<String, Double> map = new HashMap<>();
        map.put("m1", 80.0);
        map.put("m2", 75.1);
        map.put("m3", 86.2);
        map.put("m4", 64.3);
        map.put("m5", 97.4);
        assertEquals(5, jedis.zadd("k1", map));
    }

    @Test
    @DisplayName("zrange")
    public void test_02_zrange() {
        assertTrue(jedis.zrange("k1", 0, -1).containsAll(Arrays.asList("m1", "m2", "m3", "m4", "m5")));
    }

    @Test
    public void test_03_zrange_scope() {
        Set<Tuple> k1 = jedis.zrangeWithScores("k1", 0, -1);
        System.out.println("k1 = " + k1);
    }

    @Test
    public void test_04_zrevrange() {
        jedis.zrevrange("k1", 0, -1).containsAll(Arrays.asList("m1", "m2", "m3", "m4", "m5"));
    }

    @Test
    public void test_05_zrangebyscore() {
        jedis.zrangeByScore("k1", 70.0, 90.0).containsAll(Arrays.asList("m1", "m2", "m3"));
    }

    @Test
    public void test_06_zcard() {
        assertEquals(5, jedis.zcard("k1"));
    }

    @Test
    public void test_07_zcount() {
        assertEquals(3, jedis.zcount("k1", 70.0, 90.0));
    }

    @Test
    @DisplayName("zadd")
    public void test_08_zadd() {
        Map<String, Double> map = new HashMap<>();
        map.put("m2", 75.1);
        map.put("m3", 86.2);
        map.put("m4", 64.3);
        map.put("m5", 97.4);
        map.put("m6", 80.0);
        assertEquals(5, jedis.zadd("k2", map));
    }

    @Test
    public void test_09_zinterstore() {
        Long zinterstore = jedis.zinterstore("k3", "k1", "k2");
        System.out.println("zinterstore = " + zinterstore);
        Set<Tuple> k3 = jedis.zrangeWithScores("k3", 0, -1);
        System.out.println("k3 = " + k3);
    }

    @Test
    public void test_10_zunionstore() {
        Long sunionstore = jedis.zunionstore("k4", "k1", "k2");
        System.out.println("sunionstore = " + sunionstore);
        Set<Tuple> k4 = jedis.zrangeWithScores("k4", 0, -1);
        System.out.println("k4 = " + k4);
    }

    @Test
    public void test_11_zrank() {
        assertEquals(2, jedis.zrank("k1", "m1"));
    }

    @Test
    public void test_12_zrevrank() {
        assertEquals(2, jedis.zrank("k1", "m1"));
    }

    @Test
    public void test_13_zscore() {
        assertEquals(80.0, jedis.zscore("k1", "m1"));
    }

    @Test
    public void test_14_zincrby() {
        jedis.zincrby("k1", 2.0, "m1");
        assertEquals(82.0, jedis.zscore("k1", "m1"));
    }

    @Test
    public void test_15_zrem() {
        assertEquals(5, jedis.zrem("k1", "m1", "m2", "m3", "m4", "m5"));
        assertEquals(5, jedis.zrem("k2", "m6", "m2", "m3", "m4", "m5"));
    }

    @Test
    public void test_16_closePool() {
        if (jedisPool != null) {
            jedisPool.close();
        }
    }
}
