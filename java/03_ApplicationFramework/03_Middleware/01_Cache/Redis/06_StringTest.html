# StringTest

package com.it.redisdemo;

import org.junit.After;
import org.junit.Before;
import org.junit.FixMethodOrder;
import org.junit.Test;
import org.junit.jupiter.api.DisplayName;
import org.junit.runner.RunWith;
import org.junit.runners.MethodSorters;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.junit4.SpringRunner;
import redis.clients.jedis.Jedis;
import redis.clients.jedis.JedisPool;
import redis.clients.jedis.JedisPoolConfig;

import java.util.Arrays;
import java.util.List;

import static org.junit.jupiter.api.Assertions.*;

/**
 * @program redisdemo
 * @description:
 * @author: 游元博
 * @create: 2020/04/27 17:38
 */
@SpringBootTest
@DisplayName("test redis")
@FixMethodOrder(MethodSorters.NAME_ASCENDING)
@RunWith(SpringRunner.class)
public class StringTest {

    private Jedis jedis;

    @Before
    @DisplayName("init")
    public void init() {
        JedisPoolConfig jedisPoolConfig = new JedisPoolConfig();
        jedisPoolConfig.setMaxTotal(10);
        jedisPoolConfig.setMaxIdle(10);
        jedisPoolConfig.setMinIdle(10);
        jedisPoolConfig.setMaxWaitMillis(-1);
        JedisPool jedisPool = new JedisPool(jedisPoolConfig, "49.234.79.63", 6379, 0, null, 0);
        jedis = jedisPool.getResource();
    }

    @After
    @DisplayName("close")
    public void close() {
        if (jedis != null) {
            jedis.close();
        }
    }

    @Test
    @DisplayName("set")
    public void test_01_set() {
        jedis.set("k1", "v1");
    }

    @Test
    @DisplayName("get")
    public void test_02_get() {
        String value = jedis.get("k1");
        assertEquals("v1", value);
    }

    @Test
    @DisplayName("update")
    public void test_03_update() {
        jedis.set("k1", "v2");
        assertEquals("v2", jedis.get("k1"));
    }

    @Test
    @DisplayName("del")
    public void test_04_delete() {
        jedis.del("k1");
        assertNull(jedis.get("k1"));
    }

    @Test
    @DisplayName("mulSet")
    public void test_05_mulSet() {
        jedis.mset("k1", "v1", "k2", "v2");
    }

    @Test
    @DisplayName("mulGet")
    public void test_06_mulGet() {
        List<String> valueList = jedis.mget("k1", "k2");
        assertLinesMatch(Arrays.asList("v1", "v2"), valueList);
    }

    @Test
    @DisplayName("mulDel")
    public void test_07_mulDel() {
        jedis.del("k1", "k2");
       assertLinesMatch(Arrays.asList("v1","v2"), jedis.mget("k1", "k2"));
    }


}
