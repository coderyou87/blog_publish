package com.it.redisdemo;

import org.junit.After;
import org.junit.Before;
import org.junit.FixMethodOrder;
import org.junit.Test;
import org.junit.jupiter.api.DisplayName;
import org.junit.runner.RunWith;
import org.junit.runners.MethodSorters;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.junit4.SpringRunner;
import redis.clients.jedis.Jedis;
import redis.clients.jedis.JedisPool;
import redis.clients.jedis.JedisPoolConfig;

import java.lang.reflect.Array;
import java.util.Arrays;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertTrue;

/**
 * @program redisdemo
 * @description: SetTest
 * @author: 游元博
 * @create: 2020/04/28 00:06
 */
@SpringBootTest
@RunWith(SpringRunner.class)
@FixMethodOrder(value = MethodSorters.NAME_ASCENDING)
@DisplayName("set test")
public class SetTest {


    private Jedis jedis;

    @Before
    @DisplayName("init")
    public void init() {
        JedisPoolConfig jedisPoolConfig = new JedisPoolConfig();
        jedisPoolConfig.setMaxIdle(10);
        jedisPoolConfig.setMaxTotal(10);
        jedisPoolConfig.setMaxWaitMillis(-1);
        jedisPoolConfig.setMinIdle(10);
        JedisPool jedisPool = new JedisPool(jedisPoolConfig, "49.234.79.63", 6379, 0, null, 0);
        jedis = jedisPool.getResource();
    }

    @After
    @DisplayName("close")
    public void close() {
        if (jedis != null) {
            jedis.close();
        }
    }

    @Test
    @DisplayName("sadd")
    public void test_01_sadd() {
        assertEquals(5, jedis.sadd("k1", "v1", "v2", "v3", "v4", "v5"));
    }

    @Test
    @DisplayName("smembers")
    public void test_02_smembers() {
        Set<String> set = jedis.smembers("k1");
        assertTrue(set.containsAll(Arrays.asList("v1", "v2", "v3", "v4", "v5")));
    }

    @Test
    @DisplayName("srem")
    public void test_03_srem() {
        assertEquals(1, jedis.srem("k1", "v1"));
    }

    @Test
    @DisplayName("sacrd")
    public void test_04_scard() {
        assertEquals(4, jedis.scard("k1"));
    }

    @Test
    @DisplayName("sismember")
    public void test_05_sismember() {
        assertTrue(jedis.sismember("k1", "v2"));
    }

    @Test
    @DisplayName("srandmember")
    public void test_06_srandmember() {
        List<String> list = jedis.srandmember("k1", 2);
        assertTrue(Arrays.asList("v2", "v3", "v4", "v5").containsAll(list));
    }

    @Test
    @DisplayName("sadd")
    public void test_07_sadd() {
        assertEquals(5, jedis.sadd("k2", "v3", "v4", "v5", "v6", "v7"));
    }

    @Test
    @DisplayName("sinter")
    public void test_08_sinter() {
        assertTrue(jedis.sinter("k1", "k2").containsAll(Arrays.asList("v3", "v4", "v5")));
    }

    @Test
    @DisplayName("sunion")
    public void test_09_sunion() {
        assertTrue(jedis.sunion("k1", "k2").containsAll(Arrays.asList("v2", "v3", "v4", "v5", "v6", "v7")));
    }

    @Test
    @DisplayName("sdiff")
    public void test_10_sdiff() {
        assertTrue(jedis.sdiff("k1", "k2").containsAll(Arrays.asList("v2")));
    }

    @Test
    @DisplayName("sdiff")
    public void test_11_sdiff() {
        assertTrue(jedis.sdiff("k2", "k1").containsAll(Arrays.asList("v6", "v7")));
    }

    @Test
    @DisplayName("sinterstore")
    public void test_12_sinterstore() {
        assertEquals(3, jedis.sinterstore("k3", "k1", "k2"));
    }

    @Test
    @DisplayName("sunionstore")
    public void test_13_sunionstore() {
        assertEquals(6, jedis.sunionstore("k4", "k1", "k2"));
    }

    @Test
    @DisplayName("sdiffstore")
    public void test_14_sdiffstore() {
        assertEquals(1, jedis.sdiffstore("k5", "k1", "k2"));
    }

    @Test
    @DisplayName("sdiffstore")
    public void test_15_sdiffstore() {
        assertEquals(2, jedis.sdiffstore("k6", "k2", "k1"));
    }

    @Test
    @DisplayName("smove")
    public void test_16_smove() {
        assertEquals(1, jedis.smove("k1", "k7", "v2"));
    }

    @Test
    @DisplayName("spop")
    public void test_17_spop() {

        assertTrue(jedis.spop("k1", 4).containsAll(Arrays.asList("v3", "v4", "v5")));
        assertTrue(jedis.spop("k2", 5).containsAll(Arrays.asList("v3", "v4", "v5", "v6", "v7")));
        assertTrue(jedis.spop("k3", 3).containsAll(Arrays.asList("v3", "v4", "v5")));
        assertTrue(jedis.spop("k4", 6).containsAll(Arrays.asList("v2", "v3", "v4", "v5", "v6", "v7")));
        assertTrue(jedis.spop("k5", 1).contains("v2"));
        assertTrue(jedis.spop("k6", 2).containsAll(Arrays.asList("v6", "v7")));
        assertTrue(jedis.spop("k7", 1).contains("v2"));
    }

}
