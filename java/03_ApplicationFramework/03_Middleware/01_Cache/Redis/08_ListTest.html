# ListTest

package com.it.redisdemo;

import org.junit.After;
import org.junit.Before;
import org.junit.FixMethodOrder;
import org.junit.Test;
import org.junit.jupiter.api.DisplayName;
import org.junit.runner.RunWith;
import org.junit.runners.MethodSorters;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.junit4.SpringRunner;
import redis.clients.jedis.Jedis;
import redis.clients.jedis.JedisPool;
import redis.clients.jedis.JedisPoolConfig;

import java.util.Arrays;
import java.util.List;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertLinesMatch;

/**
 * @program redisdemo
 * @description: ListTest
 * @author: 游元博
 * @create: 2020/04/27 23:16
 */
@SpringBootTest
@RunWith(SpringRunner.class)
@FixMethodOrder(value = MethodSorters.NAME_ASCENDING)
@DisplayName("hash test")
public class ListTest {

    private Jedis jedis;

    @Before
    @DisplayName("init")
    public void init() {
        JedisPoolConfig jedisPoolConfig = new JedisPoolConfig();
        jedisPoolConfig.setMaxIdle(10);
        jedisPoolConfig.setMaxTotal(10);
        jedisPoolConfig.setMaxWaitMillis(-1);
        jedisPoolConfig.setMinIdle(10);
        JedisPool jedisPool = new JedisPool(jedisPoolConfig, "49.234.79.63", 6379, 0, null, 0);
        jedis = jedisPool.getResource();
    }

    @After
    @DisplayName("close")
    public void close() {
        if (jedis != null) {
            jedis.close();
        }
    }

    @Test
    @DisplayName("lpush")
    public void test_01_lpush() {
        assert 3 == jedis.lpush("k1", "v1", "v2", "v3");
    }

    @Test
    @DisplayName("lrange")
    public void test_02_lrange() {
        List<String> valueList = jedis.lrange("k1", 0, -1);
        assertLinesMatch(Arrays.asList("v3", "v2", "v1"), valueList);
    }

    @Test
    @DisplayName("lindex")
    public void test_03_lindex() {
        assertEquals("v3", jedis.lindex("k1", 0));
    }

    @Test
    @DisplayName("llen")
    public void test_04_llen() {
        assertEquals(3, jedis.llen("k1"));
    }

    @Test
    @DisplayName("lpop")
    public void test_05_lpop() {
        assertEquals("v3", jedis.lpop("k1"));
    }

    @Test
    @DisplayName("rpop")
    public void test_06_rpop() {
        assertEquals("v1", jedis.rpop("k1"));
    }

    @Test
    @DisplayName("ltrim")
    public void test_07_ltrim() {
        jedis.ltrim("k1", 1, 0);
    }
}
