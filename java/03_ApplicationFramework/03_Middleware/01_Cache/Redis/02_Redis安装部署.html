# Redis安装部署

# 一、单机

  1. 依赖
    yum -y install gcc-c++
  2. 下载
    cd /opt/software/
wget http://download.redis.io/releases/redis-5.0.5.tar.gz
  3. 解压
    tar -zxvf /opt/software/redis-5.0.5.tar.gz -C /opt/module
mv /opt/module/redis-5.0.5 /opt/module/redis
  4. 安装
    cd /opt/module/redis 
make && make install
  5. 测试
    cd /usr/local/bin
redis-server &
redis-cli
ping

# 二、主从复制

主节点可读可写，从节点只读不写。如果主节点挂了，整个集群就不能写。

  1. 安装

    tar -zxvf /opt/software/redis-5.0.5.tar.gz -C /opt/module
mv /opt/module/redis-5.0.5 /opt/module/redis
cd /opt/module/redis 
make && make install

  2. 配置 复制配置文件

    mkdir /usr/local/redis-cluster/
cp /opt/module/redis/redis.conf /usr/local/redis-cluster/
cp /usr/local/redis-cluster/redis.conf /usr/local/redis-cluster/redis-7001.conf
cp /usr/local/redis-cluster/redis.conf /usr/local/redis-cluster/redis-7002.conf
cp /usr/local/redis-cluster/redis.conf /usr/local/redis-cluster/redis-7003.conf
cp /usr/local/redis-cluster/redis.conf /usr/local/redis-cluster/redis-7004.conf

cp /usr/local/redis-cluster/redis.conf /usr/local/redis-cluster/redis-7005.conf

修改配置文件

```bash
vim /usr/local/redis-cluster/redis-7001.conf
port 7001

vim /usr/local/redis-cluster/redis-7002.conf
port 7002
replicaof 127.0.0.1 7001

vim /usr/local/redis-cluster/redis-7003.conf
port 7003
replicaof 127.0.0.1 7001
  1. 启动 启动Redis集群
    /usr/local/bin/redis-server /usr/local/redis-cluster/redis-7001.conf &
/usr/local/bin/redis-server /usr/local/redis-cluster/redis-7002.conf &
/usr/local/bin/redis-server /usr/local/redis-cluster/redis-7003.conf &
/usr/local/bin/redis-server /usr/local/redis-cluster/redis-7004.conf &
/usr/local/bin/redis-server /usr/local/redis-cluster/redis-7005.conf &
    查看Redis是否启动成功
    ps -ef | grep redis
  2. 测试
    /usr/local/bin/redis-cli -p 7001 set k1 v1
/usr/local/bin/redis-cli -p 7001 get k1 
/usr/local/bin/redis-cli -p 7002 get k1
/usr/local/bin/redis-cli -p 7003 get k1

/usr/local/bin/redis-cli -p 7002 set k2 v2
/usr/local/bin/redis-cli -p 7003 set k3 v3

/usr/local/bin/redis-cli -p 7001 set k1 value1

# 三、哨兵模式

在主从复制模式中，如果主节点宕机，则整个集群中就不能。因为从节点数据与主节点同步，当主节点宕机，将某一个从节点变为主节点，就能解决集群不能写的问题。

# 3.1 哨兵的任务

  1. 监控：哨兵会不断检查主节点与从节点服务器的运行情况

  2. 提醒：当哨兵监控的服务器出现问题时，哨兵会通过API通知管理员或其他应用程序。

  3. 自动故障迁移：当主服务器不能工作时，哨兵会在从服务器中选举一个从服务器来作为新的主服务器。如果原来的主服务器正常了，将变为从服务器。

# 3.2 安装与配置

  1. 安装

    哨兵模式的集群安装，与主从复制相似。

  2. 配置sentinel

    mkdir /usr/local/redis-sentinel
cp /opt/module/redis/sentinel.conf /usr/local/redis-sentinel/
cp /usr/local/redis-sentinel/sentinel.conf /usr/local/redis-sentinel/sentinel-7101.conf
cp /usr/local/redis-sentinel/sentinel.conf /usr/local/redis-sentinel/sentinel-7102.conf
cp /usr/local/redis-sentinel/sentinel.conf /usr/local/redis-sentinel/sentinel-7103.conf

    修改配置文件

    vim /usr/local/redis-sentinel/sentinel-7101.conf
port 7101
sentinel monitor redis7001 127.0.0.1 7001 2

    vim /usr/local/redis-sentinel/sentinel-7102.conf
port 7102
sentinel montior redis7001 127.0.0.1 7001 2

    vim /usr/local/redis-sentinel/sentinel-7103.conf
port 7103
sentinel montior redis7001 127.0.0.1 7001 2

    配置解释：

    • redis7001：主节点的别名
    • 127.0.0.1：主节点地址
    • 7001：主节点端口
    • 2：触发主节点切换的最少票数

  3. 启动哨兵

    /usr/local/bin/redis-sentinel /usr/local/redis-sentinel/sentinel-7101.conf &
/usr/local/bin/redis-sentinel /usr/local/redis-sentinel/sentinel-7102.conf &
/usr/local/bin/redis-sentinel /usr/local/redis-sentinel/sentinel-7103.conf &

  4. 查看当前各个节点的角色

    /usr/local/bin/redis-cli -p 7001 role
/usr/local/bin/redis-cli -p 7002 role
/usr/local/bin/redis-cli -p 7003 role
/usr/local/bin/redis-cli -p 7004 role
/usr/local/bin/redis-cli -p 7005 role

  5. 将主节点下线

    ps -ef | grep redis 
kill -9 XXX

  6. 查看剩下的节点角色

    /usr/local/bin/redis-cli -p 7002 role
/usr/local/bin/redis-cli -p 7003 role

  7. 上线redis7001

    /usr/local/bin/redis-server /usr/local/redis-cluster/redis-7001.conf

  8. 查看节点角色

    /usr/local/bin/redis-cli -p 7001 role
/usr/local/bin/redis-cli -p 7002 role
/usr/local/bin/redis-cli -p 7003 role

# 四、集群

在哨兵模式中，仍然只有一个主节点，当并发写请求较大时，哨兵模式不能缓解写压力。如果配置多个master节点，就能缓解写请求的压力。

集群特点：

  1. 由多个Redis服务器组成的分布式网络服务集群

  2. 集群之中有多个master主节点，每一个主节点都可读可写

  3. 节点之间相互通信，两两相连

  4. 集群无中心节点

  5. 主节点上可以连接从节点

  6. 主节点与从节点之间能够进行故障转移（由其他在线的主节点完成，不需要额外的哨兵）

  7. 分片策略，集群将整个数据库分成16384个槽位slot，数据存储在其中某一个槽位上，一个槽位可以存储多个数据。当所有槽位都有某个节点复制处理的时候，集群进入上线状态，并开始处理客户端发送的请求命令。

  8. 集群专向，由于集群没有中心节点，主节点只会处理自己负责槽位的命令请求。如果节点收到了其他槽位的命令请求，会向客户端返回一个转向错误，客户端根据错误信息中的地址与端口，向正确的槽位发送请求。
