# 原型模式

# 一、介绍

  • 原型模式（Prototype Pattern）是用于创建重复的对象，同时又能保证性能。

  • 本体给外部提供一个克隆体进行使用

    1. 数据需要缓存，从而提升性能，降低数据库压力
    2. 不能直接返回从缓存中查询到的对象，需要返回克隆对象

# 二、应用场景

  • 资源优化
  • 性能和安全要求
  • 一个对象多个修改者的场景。
  • 一个对象需要提供给其他对象访问，而且各个调用者可能都需要修改其值时可以考虑使用原型模式拷贝多个对象供调用者使用。
  • 深拷贝（两个完全对象不一样的【递归克隆】，内容却完全一样）、浅拷贝（只是属性赋值）

# 三、实现

import lombok.Data;
import lombok.ToString;

@Data
@ToString
public class User implements Cloneable {

    private String name;

    private Integer age;

    @Override
    protected Object clone() throws CloneNotSupportedException {
        User user = new User();
        user.setName(this.name);
        user.setAge(this.age);
        return user;
    }
}



import java.util.HashMap;
import java.util.Map;

public class ProtoType {
    public static void main(String[] args) throws CloneNotSupportedException {

        Map<String, User> cache = new HashMap<>();

        // 从数据库中查询数据，将查询到的数据放入缓存
        User xiaoxiao1 = getUserFromDB("xiaoxiao1", 18);
        cache.put("xiaoxiao1", xiaoxiao1);

        // 从缓存中获取数据，并返回克隆对象
        // 如果不返回克隆对象，则外部的修改，会影响缓存中的数据，导致缓存中存在脏数据
        User xiaoxiao2 = (User) cache.get("xiaoxiao1").clone();
        System.out.println("xiaoxiao2 = " + xiaoxiao2);

        // 修改缓存中的数据
        xiaoxiao2.setName("xiaoxiao2");
        System.out.println("xiaoxiao2 = " + xiaoxiao2);

        // 再次从缓存中查询数据，虽然xiaoxiao2已经被修改了，但是缓存中的数据不会被修改
        System.out.println("xiaoxiao1 = " + cache.get("xiaoxiao1"));


    }

    /**
     * 模拟从数据库查询数据
     * @param name 姓名
     * @param age 年龄
     * @return user
     */
    public static User getUserFromDB(String name, Integer age) {
        User user = new User();
        user.setName(name);
        user.setAge(age);
        return user;
    }
}
