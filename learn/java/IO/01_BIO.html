# Java-BIO

# 一、字节流

# FileInputStream

使用FileInputStream读取文件时，如果是读取中文，需要使用字节数组，以防止字符被分割。

private static void readFile1() throws IOException {
    String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\demo.txt";
    File file = new File(filePath);
    try (InputStream inputStream = new FileInputStream(file)) {
        StringBuilder stringBuilder = new StringBuilder();
        // 使用read()方法读取中文时，会出现乱码
        int i;
        while ((i = inputStream.read()) != -1) {
            stringBuilder.append((char) i);
        }
        System.out.println(stringBuilder);
    }
}

private static void readFile2() throws Exception {
    String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\demo.txt";
    File file = new File(filePath);
    try (InputStream inputStream = new FileInputStream(file)) {
        int i;
        byte[] bytes = new byte[inputStream.available()];
        while ((i = inputStream.read(bytes)) != -1) {
            System.out.println(new String(bytes, StandardCharsets.UTF_8));
        }
    }
}

private static void readFile3() throws Exception {
    String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\demo.txt";
    File file = new File(filePath);
    try (InputStream inputStream = new FileInputStream(file)) {
        byte[] bytes = new byte[inputStream.available()];
        while (inputStream.read(bytes, 0, bytes.length) != -1) {
            System.out.println(new String(bytes, StandardCharsets.UTF_8));
        }
    }
}

# FileOutputStream

private static void writeFile1() throws Exception {
        String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\write.txt";
        File file = new File(filePath);
        String content = "a,b,c,d,e,f";
        try (OutputStream outputStream = new FileOutputStream(file)) {
            byte[] bytes = content.getBytes();
            outputStream.write(bytes);
            outputStream.write("\n".getBytes());
            outputStream.write("1,2,3,4,5,6".getBytes());
            outputStream.flush();
        }
    }

    private static void writeFile2() throws Exception {
        String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\write.txt";
        File file = new File(filePath);
        try (OutputStream outputStream = new FileOutputStream(file)) {
            for (int i = 0; i < 10; i++) {
                outputStream.write(String.valueOf(i).getBytes());
                outputStream.write("\n".getBytes());
            }
            outputStream.flush();
        }
    }

    private static void writeFile3() throws Exception {
        String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\write.txt";
        File file = new File(filePath);
        try (OutputStream outputStream = new FileOutputStream(file)) {
            for (int i = 0; i < 10; i++) {
                outputStream.write(("小" + i + "\n").getBytes());
            }
        }
    }

# FileReader

private static void readFile1() throws Exception {
        String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\demo.txt";
        File file = new File(filePath);
        try (InputStreamReader inputStreamReader = new FileReader(file);) {
            char[] chars = new char[1024];
            while (inputStreamReader.read(chars, 0, 1023) != -1) {
                System.out.println(new String(chars));
            }
        }
    }

    private static void readFile2() throws Exception {
        String filePath = "E:\\Workspace\\study\\java-io\\src\\com\\it\\stream\\input\\demo.txt";
        File file = new File(filePath);
        try (Reader reader = new FileReader(file);) {
            char[] chars = new char[1024];
            while (reader.ready()){
                reader.read(chars);
                System.out.println(chars);
            }
        }
    }

# BufferedReader

private static void readFile1() throws Exception {
        try (Reader reader = new InputStreamReader(new FileInputStream(new File(filePath)));
             BufferedReader bufferedReader = new BufferedReader(reader)) {

            char[] chars = new char[1024];
            while (bufferedReader.ready()) {
                bufferedReader.read(chars);
                System.out.println(chars);
            }
        }
    }

    private static void readFile2() throws Exception {
        try (BufferedReader bufferedReader = new BufferedReader(new FileReader(filePath))) {

            }
        }
    }

    private static void readFile3() throws Exception {

        }
    }
