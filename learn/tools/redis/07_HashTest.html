# HashTest

package com.it.redisdemo;

import org.junit.After;
import org.junit.Before;
import org.junit.FixMethodOrder;
import org.junit.Test;
import org.junit.jupiter.api.DisplayName;
import org.junit.runner.RunWith;
import org.junit.runners.MethodSorters;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.junit4.SpringRunner;
import redis.clients.jedis.Jedis;
import redis.clients.jedis.JedisPool;
import redis.clients.jedis.JedisPoolConfig;

import java.util.*;

import static org.junit.jupiter.api.Assertions.*;

/**
 * @program redisdemo
 * @description: HashTest
 * @author: 游元博
 * @create: 2020/04/27 19:35
 */
@SpringBootTest
@DisplayName("test redis")
@FixMethodOrder(MethodSorters.NAME_ASCENDING)
@RunWith(SpringRunner.class)
public class HashTest {
    private Jedis jedis;

    @Before
    @DisplayName("init")
    public void init() {
        JedisPoolConfig jedisPoolConfig = new JedisPoolConfig();
        jedisPoolConfig.setMaxTotal(10);
        jedisPoolConfig.setMaxIdle(10);
        jedisPoolConfig.setMinIdle(10);
        jedisPoolConfig.setMaxWaitMillis(-1);
        JedisPool jedisPool = new JedisPool(jedisPoolConfig, "49.234.79.63", 6379, 0, null, 0);
        jedis = jedisPool.getResource();
    }

    @After
    @DisplayName("close")
    public void close() {
        if (jedis != null) {
            jedis.close();
        }
    }

    @Test
    @DisplayName("hset")
    public void test_01_hset() {
        Map<String, String> map = new HashMap<>();
        map.put("f1", "v1");
        map.put("f2", "v2");
        jedis.hset("k1", map);
    }

    @Test
    @DisplayName("hget")
    public void test_02_hget() {
        String v1 = jedis.hget("k1", "f1");
        String v2 = jedis.hget("k1", "f2");
        assertEquals("v1", v1);
        assertEquals("v2", v2);
    }

    @Test
    @DisplayName("hgetall")
    public void test_03_hgetall() {
        Map<String, String> valueMap = jedis.hgetAll("k1");
        assert valueMap.containsKey("f1");
        assert valueMap.containsValue("v1");

        assert valueMap.containsKey("f2");
        assert valueMap.containsValue("v2");
    }

    @Test
    @DisplayName("hmset")
    public void test_04_hmset() {
        Map<String, String> map = new HashMap<>();
        map.put("f3", "v3");
        map.put("f4", "v4");
        jedis.hmset("k2", map);
    }

    @Test
    @DisplayName("hmget")
    public void test_05_hmget() {
        List<String> valueList = jedis.hmget("k2", "f3", "f4");
        assertLinesMatch(Arrays.asList("v3", "v4"), valueList);
    }

    @Test
    @DisplayName("hlen")
    public void test_06_hlen() {
        Long hlen = jedis.hlen("k2");
        assertEquals(2L, hlen);
    }

    @Test
    @DisplayName("hexists")
    public void test_07_hexists() {
        Boolean hexists = jedis.hexists("k2", "f3");
        assertTrue(hexists);
    }

    @Test
    @DisplayName("hkeys")
    public void test_08_hkeys() {
        Set<String> hkeys = jedis.hkeys("k2");
        //assertLinesMatch(Arrays.asList("f3", "f4"), new ArrayList<>(hkeys));
        assert hkeys.containsAll(Arrays.asList("f3", "f4"));
    }

    @Test
    @DisplayName("hvals")
    public void test_09_hvals() {
        List<String> valueList = jedis.hvals("k2");
        //assertLinesMatch(Arrays.asList("v3", "v4"), valueList);
        assert valueList.containsAll(Arrays.asList("v3", "v4"));
    }

    @Test
    @DisplayName("hdel")
    public void test_10_hdel() {
        assertEquals(2L, jedis.hdel("k1", "f1", "f2"));
        assertEquals(2L, jedis.hdel("k2", "f3", "f4"));
    }

}
