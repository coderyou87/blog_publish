  5. 创建配置类

    CarDatasourceConfig:

    package com.it.multidata.config;

import com.zaxxer.hikari.HikariDataSource;
import org.apache.ibatis.session.SqlSessionFactory;
import org.mybatis.spring.SqlSessionFactoryBean;
import org.mybatis.spring.annotation.MapperScan;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.context.annotation.Primary;
import org.springframework.core.io.Resource;
import org.springframework.core.io.support.PathMatchingResourcePatternResolver;
import org.springframework.jdbc.datasource.DataSourceTransactionManager;

import javax.sql.DataSource;

@Configuration
@MapperScan(basePackages = CarDatasourceConfig.PACKAGE, sqlSessionFactoryRef = "carSqlSessionFactory")
public class CarDatasourceConfig {
    public static final String PACKAGE = "com.it.multidata.car.mapper";
    public static final String MAPPER_LOCATION = "classpath:mapper/car/*.xml";

    @Value("${spring.datasource.test_data_car.url}")
    private String url;

    @Value("${spring.datasource.test_data_car.driver-class-name}")
    private String driverClassName;

    @Value("${spring.datasource.test_data_car.username}")
    private String username;

    @Value("${spring.datasource.test_data_car.password}")
    private String password;

    @Bean("carDataSource")
    @Primary
    public DataSource carDataSource() {
        HikariDataSource hikariDataSource = new HikariDataSource();
        hikariDataSource.setJdbcUrl(url);
        hikariDataSource.setDriverClassName(driverClassName);
        hikariDataSource.setUsername(username);
        hikariDataSource.setPassword(password);
        return hikariDataSource;
    }

    @Bean(name = "carTransactionManager")
    @Primary
    public DataSourceTransactionManager carTransactionManager() {
        return new DataSourceTransactionManager(carDataSource());
    }

    @Bean(name = "carSqlSessionFactory")
    @Primary
    public SqlSessionFactory carSqlSessionFactory(@Qualifier("carDataSource") DataSource dataSource) throws Exception {
        SqlSessionFactoryBean sqlSessionFactoryBean = new SqlSessionFactoryBean();
        sqlSessionFactoryBean.setDataSource(dataSource);
        PathMatchingResourcePatternResolver pathMatchingResourcePatternResolver = new PathMatchingResourcePatternResolver();
        Resource[] resources = pathMatchingResourcePatternResolver.getResources(CarDatasourceConfig.MAPPER_LOCATION);
        sqlSessionFactoryBean.setMapperLocations(resources);
        return sqlSessionFactoryBean.getObject();
    }
}

  7. 编写Mapper接口和Mapper.xml

    CarMapper接口：

    package com.it.multidata.car.mapper;

import com.it.multidata.car.domain.Car;
import org.apache.ibatis.annotations.Param;

import java.util.List;

public interface CarMapper {

    void addCat(Car car);

    Car queryCarById(@Param("id") String id);

    List<Car> queryAllCar();
}

    CarMapper.xml：

    <?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd" >
<mapper namespace="com.it.multidata.car.mapper.CarMapper">

    <insert id="addCat">
        insert into car (id, name, brand, type)
        values (#{id}, #{name}, #{brand}, #{type})
    </insert>

    <select id="queryCarById" resultType="com.it.multidata.car.domain.Car">
        select id, name, brand, type
        from car
        where id = #{id}
    </select>

    <select id="queryAllCar" resultType="com.it.multidata.car.domain.Car">
        select id, name, brand, type
        from car
    </select>
</mapper>

  8. 测试

    package com.it.multidata;

import com.it.multidata.car.domain.Car;
import com.it.multidata.car.mapper.CarMapper;
import com.it.multidata.person.domain.Person;
import com.it.multidata.person.mapper.PersonMapper;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;

import java.util.List;
import java.util.UUID;

@SpringBootTest
class MultidataApplicationTests {

    @Autowired
    private CarMapper carMapper;

    @Autowired
    private PersonMapper personMapper;

    @Test
    void test01() {
        Car car = new Car();
        car.setId(UUID.randomUUID().toString());
        car.setName("haha");
        car.setBrand("奔驰");
        car.setType("小车");
        carMapper.addCat(car);
    }

    @Test
    void test02() {
        List<Car> carList = carMapper.queryAllCar();
        System.out.println(carList);
        assert carList.size() == 1;
    }

    @Test
    void test03() {
        Car car = carMapper.queryCarById("97f63266-f46f-4a01-a3bc-f2927c79ac57");
        assert car != null;
    }

    @Test
    void test04() {
        Person person = new Person();
        person.setId(UUID.randomUUID().toString());
        person.setName("xiaoxiao");
        person.setSex("man");
        person.setAddress("chengdu");
        int result = personMapper.addPerson(person);
        assert result == 1;
    }

    @Test
    void test05() {
        List<Person> personList = personMapper.queryAllPerson();
        System.out.println(personList);
        assert personList.size() == 1;
    }

    @Test
    void test06() {
        Person person = personMapper.queryPersonById("2a28bc5d-678c-442c-b03c-ecfe79f5e19b");
        assert person != null;
    }
}
