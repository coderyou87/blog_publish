# 一、 取得每个部门最高薪水的人员名称

  1. 查询每个部门的最高薪水

    select e.deptno, max(e.sal) maxsal
from emp e
group by e.deptno;

  2. 将查询的结果作为临时表t，使用emp表与t表进行关联（关联条件：deptno相同，sal等于maxSal）

  3. 查询人员名称

    select deptno, ename, sal, sal maxsal
from emp
where (deptno, sal) in
      (select e.deptno, max(e.sal) maxsal
       from emp e
       group by e.deptno)
order by deptno;

    select e.deptno, e.ename, e.sal, t.maxsal
from (select e.deptno, max(e.sal) maxsal
      from emp e
      group by e.deptno) t
         inner join emp e
                    on t.deptno = e.deptno
where t.maxsal = e.sal
order by deptno;

# 二、 哪些人的薪水在部门平均薪水之上

  1. 查询每个部门的平均薪水

    select deptno, avg(sal)
from emp
group by deptno;

  2. 将查询结果作为临时表 t，使用emp表与t表进行关联（关联条件：deptno相同，sal大于avgSal）

    select emp.*
from (select deptno, avg(sal) avgSal
      from emp
      group by deptno) t
         inner join emp
                    on t.deptno = emp.deptno
where sal > t.avgSal;

# 三、取得部门中（所有人的）平均薪水等级

  1. 查询每个部门的平均薪水

    select deptno, avg(sal)
from emp
group by deptno;

  2. 将查询结果作为临时表t，与salgrade表进行关联（关联条件，avgSal在salgrade表的losal与hisal之间）

    select deptno, avgSal, salgrade.grade
from (select deptno, avg(sal) avgSal
      from emp
      group by deptno) t
         inner join salgrade
                    on t.avgSal between salgrade.losal and salgrade.hisal
order by deptno;

# 四、不准用组函数（MAX），取得最高薪水（给出两种解决方案）

  1. 查询sal字段，使用order by排序，取第一个

    select sal
from emp
order by sal desc
limit 0,1;

  2. 使用两次emp表，做内连接，关联条件是：a.sal < b.sal

    这样查询到的值，包含除最大值以外的其他值

    select distinct a.sal
from emp a
         inner join emp b
                    on a.sal < b.sal;

  3. 使用一个not in，将最大值以外的其他值排除掉，即可得到最大值

    select sal
from emp
where sal not in (select distinct a.sal
                  from emp a
                           inner join emp b
                                      on a.sal < b.sal);

# 五、取得平均薪水最高的部门的部门编号（至少两个方案）

  1. 查询deptno与avgSal，根据avgSal排序，取第一个

    select deptno, avg(sal) avgSal
from emp
group by deptno
order by avgSal desc
limit 0,1;

  2. 查询出部门平均薪水的最大值，再做等值查询

    查询出部门平均薪水的最大值

    select avg(sal) avgSal
from emp
group by deptno
order by avgSal desc
limit 0,1;

    等值查询

    select deptno, avg(sal) avgSal
from emp
group by deptno
having avgSal = (select avg(sal) avgSal
                 from emp
                 group by deptno
                 order by avgSal desc
                 limit 0,1);

# 六、取得平均薪水最高的部门的部门名称

  1. 查询部门平均薪水的最大值

    select deptno, avg(sal) avgSal
from emp
group by deptno
order by avgSal desc
limit 0,1;

  2. 查询结果与emp表做关联查询

    select dept.deptno, dept.dname, t.avgSal
from (select deptno, avg(sal) avgSal
      from emp
      group by deptno
      order by avgSal desc
      limit 0,1) t
         inner join dept on dept.deptno = t.deptno;

    select dept.deptno, dept.dname, t.avgSal
from (select deptno, avg(sal) avgSal
      from emp
      group by deptno
      order by avgSal desc
      limit 0,1) t,
     dept
where dept.deptno = t.deptno;

  3. emp表与dept表做关联查询，部门平均薪水最大值做条件

    select d.deptno, d.dname, avg(sal) avgSal
from emp e
         inner join dept d on e.deptno = d.deptno
group by d.dname
having avgSal = (select avg(sal) avgSal
                 from emp
                 group by deptno
                 order by avgSal desc
                 limit 0,1)

# 七、求平均薪水的等级最低的部门的部门名称

  1. 查询每个部门的平均薪水

    select deptno, avg(sal) avgSal
from emp
group by deptno;

  2. 查询每个部门的平均薪水等级

    select deptno, avgSal, grade
from (select deptno, avg(sal) avgSal
      from emp
      group by deptno) t,
     salgrade
where t.avgSal between salgrade.losal and salgrade.hisal;

  3. 查询平均薪水等级最低的deptno，avgSal，grade

    select deptno, avgSal, grade
from (select deptno, avg(sal) avgSal
      from emp
      group by deptno) t,
     salgrade
where t.avgSal between salgrade.losal and salgrade.hisal
order by grade
limit 0,1;

  4. 求平均薪水的等级最低的部门的部门名称

    select dept.deptno, dept.dname, t2.grade
from (select deptno, avgSal, grade
      from (select deptno, avg(sal) avgSal
            from emp
            group by deptno) t,
           salgrade
      where t.avgSal between salgrade.losal and salgrade.hisal
      order by grade
      limit 0,1) t2,
     dept
where t2.deptno = dept.deptno;

# 十八、列出所有“CLERK”（办事员）的姓名及其部门名称，部门人数

      from emp,

      where emp.job = 'CLERK'
        and emp.deptno = dept.deptno) t1
         inner join (select deptno, count(deptno) emp_count
                     from emp
                     group by deptno) t2
                    on t1.deptno = t2.deptno;

where emp.deptno = dept.deptno
  and dept.deptno = t.deptno
  and emp.job = 'CLERK';
